## PROPS INTERFACE
```typescript
interface RelatedNotesTabProps {
  noteId: string;
  relatedNotes: RelatedNote[];
  referencedNotes: ReferencedNote[];
  isLoading: boolean;
  sortBy: 'relevance' | 'recent' | 'alphabetical';
  filterBy: 'all' | 'sameFolder' | 'tagged' | 'recent';
  onNoteClick: (noteId: string) => void;
  onAddReference: (noteId: string) => void;
  onRemoveReference: (noteId: string) => void;
  onHideSuggestion: (noteId: string) => void;
  onRefresh: () => void;
  onSortChange: (sort: string) => void;
  onFilterChange: (filter: string) => void;
  onShowMore: () => void;
}

interface RelatedNote {
  id: string;
  noteId: string;
  title: string;
  icon?: string;
  similarity: number;           // 0-100
  reason: string;               // Why it's related
  folder?: string;
  updatedAt: Date;
  isHidden?: boolean;
}

interface ReferencedNote {
  id: string;
  noteId: string;
  title: string;
  icon?: string;
}
```

```javascript
const getPercentageColor = (percent) => {
  if (percent >= 90) return { text: '#16a34a', bg: '#dcfce7' }; // koyu yeşil
  if (percent >= 80) return { text: '#22c55e', bg: '#dcfce7' }; // yeşil
  if (percent >= 70) return { text: '#84cc16', bg: '#ecfccb' }; // lime
  if (percent >= 60) return { text: '#eab308', bg: '#fef9c3' }; // sarı
  if (percent >= 50) return { text: '#f97316', bg: '#ffedd5' }; // turuncu
  return { text: '#ef4444', bg: '#fee2e2' }; // kırmızı
};

const getPercentageLabel = (percent) => {
  if (percent >= 90) return 'Çok Yüksek';
  if (percent >= 80) return 'Yüksek';
  if (percent >= 70) return 'İyi';
  if (percent >= 60) return 'Orta';
  if (percent >= 50) return 'Düşük';
  return 'Çok Düşük';
};
```
