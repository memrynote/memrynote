### T031: JWT Validation Middleware

**File**: `sync-server/src/middleware/auth.ts`

**Implementation**:

```typescript
import { createMiddleware } from 'hono/factory'
import * as jose from 'jose'

// JWT configuration
const JWT_ALGORITHM = 'HS256' // or 'EdDSA' for Ed25519
const JWT_EXPIRY = '15m'

// Middleware to validate JWT in Authorization header
export const jwtAuth = createMiddleware(async (c, next) => {
  const authHeader = c.req.header('Authorization')
  if (!authHeader?.startsWith('Bearer ')) {
    return c.json({ error: 'AUTH_TOKEN_MISSING' }, 401)
  }

  const token = authHeader.slice(7)
  try {
    const { payload } = await jose.jwtVerify(token, secret, {
      algorithms: [JWT_ALGORITHM]
    })
    c.set('user', payload)
    c.set('userId', payload.sub)
    c.set('deviceId', payload.device_id)
    await next()
  } catch (error) {
    return c.json({ error: 'AUTH_TOKEN_INVALID' }, 401)
  }
})
```

**Key considerations**:

- Use `jose` library for Workers-compatible JWT
- Extract user_id and device_id from token claims
- Store in Hono context for downstream handlers

- On rotation: set `rotated_at`, issue new token
- On revoke: set `revoked = 1` for all device tokens

### T032: Rate Limiting Middleware

**File**: `sync-server/src/middleware/rate-limit.ts`

**Implementation**:

```typescript
import { createMiddleware } from 'hono/factory'

interface RateLimitConfig {
  requests: number // Max requests
  window: number // Window in seconds
  keyPrefix: string // e.g., 'otp', 'auth', 'sync'
}

// Rate limit configs from contracts
export const RATE_LIMITS = {
  OTP_REQUEST: { requests: 3, window: 600, keyPrefix: 'otp' },
  OTP_VERIFY: { requests: 5, window: 600, keyPrefix: 'otp-verify' },
  TOKEN_REFRESH: { requests: 30, window: 60, keyPrefix: 'refresh' },
  SYNC_PUSH: { requests: 100, window: 60, keyPrefix: 'sync-push' }
}

export const rateLimit = (config: RateLimitConfig) =>
  createMiddleware(async (c, next) => {
    const key = `${config.keyPrefix}:${extractKey(c)}`
    const now = Math.floor(Date.now() / 1000)

    // Check/update rate limit in D1
    const result = await c.env.DB.prepare(
      `
      INSERT INTO rate_limits (id, key, count, window_start)
      VALUES (?, ?, 1, ?)
      ON CONFLICT(key) DO UPDATE SET
        count = CASE 
          WHEN window_start + ? <= ? THEN 1 
          ELSE count + 1 
        END,
        window_start = CASE 
          WHEN window_start + ? <= ? THEN ? 
          ELSE window_start 
        END
      RETURNING count, window_start
    `
    )
      .bind(uuid(), key, now, config.window, now, config.window, now, now)
      .first()

    if (result && result.count > config.requests) {
      const retryAfter = result.window_start + config.window - now
      c.header('Retry-After', String(retryAfter))
      return c.json({ error: 'RATE_LIMIT_EXCEEDED' }, 429)
    }

    await next()
  })
```

### T033b: Atomic Server Cursor Increment

**File**: `sync-server/src/services/cursor.ts`

**Implementation**:

```typescript
/**
 * Atomically increment and return the next server_cursor for a user.
 * Uses D1 transaction to ensure no race conditions.
 */
export async function getNextCursor(db: D1Database, userId: string): Promise<number> {
  // Atomic increment using D1 RETURNING clause
  const result = await db
    .prepare(
      `
    UPDATE server_cursor_sequence 
    SET current_cursor = current_cursor + 1 
    WHERE user_id = ?
    RETURNING current_cursor
  `
    )
    .bind(userId)
    .first<{ current_cursor: number }>()

  if (!result) {
    // User doesn't have cursor sequence yet, create it
    await db
      .prepare(
        `
      INSERT INTO server_cursor_sequence (user_id, current_cursor)
      VALUES (?, 1)
    `
      )
      .bind(userId)
      .run()
    return 1
  }

  return result.current_cursor
}

/**
 * Get current cursor without incrementing (for sync status)
 */
export async function getCurrentCursor(db: D1Database, userId: string): Promise<number> {
  const result = await db
    .prepare(
      `
    SELECT current_cursor FROM server_cursor_sequence WHERE user_id = ?
  `
    )
    .bind(userId)
    .first<{ current_cursor: number }>()

  return result?.current_cursor ?? 0
}
```

```typescript
/**
 * Delete expired OTP codes (T034a)
 * Should run periodically (e.g., every 5 minutes via Cron Trigger)
 */
export async function cleanupExpiredOtpCodes(db: D1Database): Promise<number> {
  const now = Math.floor(Date.now() / 1000)

  const result = await db
    .prepare(
      `
    DELETE FROM otp_codes 
    WHERE expires_at < ? OR used = 1
  `
    )
    .bind(now)
    .run()

  return result.meta.changes ?? 0
}

/**
 * Delete expired linking sessions (T034b)
 * Should run periodically (e.g., every 5 minutes via Cron Trigger)
 */
export async function cleanupExpiredLinkingSessions(db: D1Database): Promise<number> {
  const now = Math.floor(Date.now() / 1000)

  const result = await db
    .prepare(
      `
    DELETE FROM linking_sessions 
    WHERE expires_at < ? AND status != 'completed'
  `
    )
    .bind(now)
    .run()

  return result.meta.changes ?? 0
}

/**
 * Run all cleanup jobs
 */
export async function runCleanupJobs(db: D1Database): Promise<{
  otpCodes: number
  linkingSessions: number
}> {
  const [otpCodes, linkingSessions] = await Promise.all([
    cleanupExpiredOtpCodes(db),
    cleanupExpiredLinkingSessions(db)
  ])

  return { otpCodes, linkingSessions }
}
```
